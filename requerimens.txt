import streamlit as st

# Función para la conversión
def celsius_a_fahrenheit(celsius: float) -> float:
    return (celsius * 9/5) + 32

# Título de la aplicación
st.title("🌡️ Convertidor de Temperatura: Celsius ➝ Fahrenheit")

# Entrada del usuario
celsius = st.number_input("Ingrese la temperatura en °C:", value=0.0, step=0.1)

# Botón para convertir
if st.button("Convertir"):
    fahrenheit = celsius_a_fahrenheit(celsius)
    st.success(f"{celsius:.2f} °C equivalen a {fahrenheit:.2f} °F")
streamlit run app.py
